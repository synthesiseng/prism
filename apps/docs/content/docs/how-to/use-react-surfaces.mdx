---
title: Use React Surfaces
description: Render React-authored HTML/CSS components as Prism-managed canvas surfaces.
---

React can own the HTML component. Prism should still own the runtime surface lifecycle.

```tsx
import { useEffect, useRef } from "react";
import { CanvasRuntime, type CanvasSurface } from "@synthesisengineering/prism";

export function CardSurface({ runtime }: { runtime: CanvasRuntime }) {
  const ref = useRef<HTMLDivElement>(null);
  const surface = useRef<CanvasSurface | null>(null);

  useEffect(() => {
    const element = ref.current;
    if (!element) return;

    surface.current = runtime.registerSurface(element, {
      bounds: { x: 40, y: 40, width: 320, height: 180 }
    });

    runtime.invalidate();

    return () => {
      surface.current?.dispose();
      surface.current = null;
      runtime.invalidate();
    };
  }, [runtime]);

  return <div ref={ref}>React-authored HTML surface</div>;
}
```

Keep React state serializable. Do not store `CanvasRuntime` or `CanvasSurface` inside your document model.
